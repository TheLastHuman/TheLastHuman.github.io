CREATE OR REPLACE TYPE "TYPE_SPLIT" IS TABLE OF VARCHAR2(4000);

定义函数

create or replace function split(var_str in string,var_split in string)
    return type_split
    PIPELINED as
    var_tmp varchar2(4000);
    var_element varchar2(4000);
    n_length Number := length(var_split);
begin
    var_tmp := var_str;
    while instr(var_tmp,var_split) > 0 loop
      var_element := substr(var_tmp,1,instr(var_split)-1);
      var_tmp :=substr(var_tmp,instr(var_tmp,var_split) + n_lenth,length(var_tmp));
      pipe row(var_element);
    end loop;
    pipe row(var_tmp);
    return;
end split;

利用游标和oracle自带递归函数实现权限控制

定义类型type_deptid

CREATE OR REPLACE TYPE type_roleid IS TABLE OF VARCHAR2(1000);

定义函数

create or replace function f_base_roleid(roleid1 in string)
    return type_roleid
    PIPELINED as
    CURSOR cursor_getrole(did varchar2) IS
    select roleid form base_role 
    start with roleid in 
    --parentid = '0' 为最顶层
    (select distinct roleid from base_role where parentid = '0' start with roleid = did connect by prior parentid = roleid)
    connect by parentid = prior roleid;
    row_data cursor_getrole%ROWTYPE;
begin
    OPEN cursor_getrole(roleid1);
        loop
            fetch cursor_getrole
                into row_data;
                pipe row(row_data.roleid);
            exit when cursor_getrole%NOTFOUND;
        end loop;
        close cursor_getrole;
    return;
end f_base_roleid;

例子

select * from roleid in (select * from f_base_roleid(roleid1));
