package com.base.util;

                            //getClassTypeValue(fields[i].getType(),value)});
                        //}
                        if(value != null){
                            method.invoke(obj,new Object[]{
                            getClassTypeValue(fields[i].getType(),value)});

    /**
     * 通过class类型获取对应类型的值
     */
    public static Object getClassTypeValue(Class<?> typeClass,String value)
    throws ParseException{
        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
        if(typeClass == Integer.calss || typeClass == int.class){
            if(StringUtils.isEmpty(value)){
                return 0;
            }   
            return Integer.parseInt(value.toString());
        }else if(typeClass == short.calss){
            if(StringUtils.isEmpty(value)){
                return 0;
            }   
            return Short.parseShort(value.toString());
        }else if(typeClass == byte.calss){
            if(StringUtils.isEmpty(value)){
                return 0;
            }   
            return Byte.parseByte(value.toString());
        }else if(typeClass == Double.calss || typeClass == double.calss){
            if(StringUtils.isEmpty(value)){
                return 0.0;
            }   
            return Double.parseDouble(value.toString());
        }else if(typeClass == long.calss){
            if(StringUtils.isEmpty(value)){
                return 0.0;
            }   
            return Long.parseLong(value.toString());
        }else if(typeClass == String.calss){
            if(StringUtils.isEmpty(value)){
                return "";
            }   
            return value.toString();
        }else if(typeClass == BigDecimal.calss){
            if(StringUtils.isEmpty(value)){
                return new BigDecimal(0);
            }   
            return new BigDecimal(value+"");
        }else if(typeClass == Date.calss){
            if(StringUtils.isEmpty(value)){
                return null;
            }   
            return sdf.parse(value.toString());
        }else{
            return typeClass.cast(value);
        }
    }
}

     * 获取两个对象相同属性不同的值的列
     */
    public static List<Map<String,Object>> compareTwoClass(Object class1,Object class2)
    throws ClassNotFoundException,IllegalAccessException{
        List<Map<String,Object>> list = new ArrayList<Map<String,Object>>();
        //获取对象的class
        Class<?> clazz1 = class1.getClass();
        Class<?> clazz2 = class2.getClass();
        //获取对象的属性列表
        Field[] field1 = clazz1.getDeclaredFields();
        Field[] field2 = clazz2.getDeclaredFields();
        //遍历属性列表field1
        for(int i=0;i<field1.length;i++){
            //遍历属性列表field2
            for(int j=0;j<field2.length;j++){
            //如果field1[i]属性名与field2[i]属性名内容相同
                if(field1[i].getName().equals(field2[j].getName())){
                    if(filterStr(field1[i].getName())){
                        field1[i].setAccessible(true);
                        field2[j].setAccessible(true);
                        //如果field1[i]属性值与field2[i]属性值内容不相同
                        if(!compareTwo(field1[i].get(class1),field2[j].get(class2))){
                            Map<String,Object> map2 = new HashMap<String,Object>();
                            map2.put("name",field1[i].getName());
                            map2.put("old",field1[i].get(class1));
                            map2.put("new",field1[2].get(class1));
                            list.add(map2);
                        }
                        break;
                    }
                }
            }
        }
        return list;
    }

    /**
     * 对比两个数据是否相同
     */
    public static boolean compareTwo(Object object1,Object object2){
        if(object1==null&&object2==null){
            return true;
        }
        if(object1==null&&object2!=null){
            return false;
        }
        if(object1.equals(object2)){
            return true;
        }
        return false;
    }

    /**
     * 筛选出不需要判断的属性
     */
    public static boolean filterStr(String filterStr){
        if(filterStr.equals("column1")||filterStr.equals("column2")){
            return false;
        }
        return true;
    }

private static Object invokeSetMethod(Object clazz,String fieldName,Object[] args){
    String methodName = fieldName.substring(0,1).toUpperCase()+fieldName.substring(1);
    Method method = null;
    try{
        Class[] parameterTypes = new Class[1];
        Class c = Class.forName(clazz.getClass().getName());
        Field field = c.getDeclaredField(fieldName);
        parameterTypes[0] = field.getType();
        method = c.getDeclaredMethod("set"+methodName,parameterTypes);
        return method.invoke(clazz,args);

        e.printStackTrace();
        return "";
