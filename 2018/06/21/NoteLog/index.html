该方法利用spring aop的切点和切面，自定义注解实现对操作日志和异常日志的捕捉，并记录到数据库。

定义注解，分别作用于controller和service拦截

定义controller层操作日志注解

package com.base.annotation
import java.lang.annotation.Documented;
import java.lang.annotation.ElementType;
import java.lang.annotation.Relention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;
/**
 * 自定义注解 controller层拦截
 */
@Target({ElementType.PARAMETER,ElementType.METHOD})
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface SystemControllerLog{
    String description() default "";
}

定义service层异常日志注解

package com.base.annotation
import java.lang.annotation.Documented;
import java.lang.annotation.ElementType;
import java.lang.annotation.Relention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;
/**
 * 自定义注解 service层拦截
 */
@Target({ElementType.PARAMETER,ElementType.METHOD})
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface SystemServiceLog{
    String description() default "";
}

定义切点类

package com.base.annotation
import java.lang.reflect.Method;
import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.annotation.Aspect;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import com.base.Log;
import com.base.LogService;

/**
 * 切点类，这边有两种方法。1.利用注解注入切点。2.通过xml托管切点
 */
@Aspect
public class SystemLogAspect{
    @Resource
    private LogService LogService;
    private static final Logger logger = LoggerFactory.getLogger(SystemLogAspect.class);

    //注解注入切点 Controller层
    //@Pointcut("@annotation(com.base.annotation.SystemControllerLog)")
    //public void controllerpot(){
    //}

    //注解注入切点 Service层
    //@Pointcut("@annotation(com.base.annotation.SystemServiceLog)")
    //public void servicepot(){
    //}

    /**
     * 前置通知 Controller层
     */
    //@Before("controllerpot()")//这里也可以使用注解来注入切点
    public void beforeMet(JoinPoint joinPoint){
        try{
            String sString = JoinPoint.getTarget().getClass().getName()
                           +"."+joinPoint.getSignature.getName()+"()";
            Log log = new Log();
            //填入日志信息到对象
            log.setExecuteresultjson(getControllerMethodDescription(JoinPoint));
            log.setDescription(sString);
            //保存到数据库
            LogService.save(log);
        }catch(Exception e){
            //记录本地异常日志
            logger.error("----前置通知异常----");
            logger.error("异常信息：{}",e);
        }
    }

    /**
     * 异常通知 Service层
     */
    //这里也可以使用注解来注入切点
    //@AfterThrowing(pointcut="servicepot()",throwing="e")
    public void afterThrowingMet(JoinPoint joinPoint,Throwable e ){
        try{
            String sString = JoinPoint.getTarget().getClass().getName()
                           +"."+joinPoint.getSignature.getName()+"()";
            Log log = new Log();
            //填入日志信息到对象
            log.setExecuteresultjson(getServiceMethodDescription(JoinPoint));
            log.setDescription(sString+"\r\n"+e.getClass().getName()
                              +":"+e.getMessage);
            //保存到数据库
            LogService.save(log);
        }catch(Exception e){
            //记录本地异常日志
            logger.error("----异常通知异常----");
            logger.error("异常信息：{}",e);
        }
    }

    /**
     * 获取controller层 joinPoint信息
     */
    public static String getControllerMethodDescription(JoinPoint joinPoint)
     throws Exception{
        //获取目标类名
        String targetName = joinPoint.getTarget().getClass().getName();
        //获取方法名
        String methodName = joinpoint.getSignature().getName();
        //获取相关参数
        Object[] arguments = joinPoint.getArgs();
        //生成类对象
        Class targetClass = Class.forName(targetName);
        //获取该类中的方法
        Method[] methods = targetClass.getMethods();

        String description = "";
        for(Method method : methods){
            if(!method.getName().equals(methodName)){
                continue;
            }
            Class[] clazz = method.getParameterTypes();
            if(clazz.length != arguments.length){
                continue;
            }
            description = method.getAnnotation(SystemControllerLog.class).description();
        }
        return description;
    }

    /**
     * 获取service层 joinPoint信息
     */
    public static String getServiceMethodDescription(JoinPoint joinPoint)
     throws Exception{
        //获取目标类名
        String targetName = joinPoint.getTarget().getClass().getName();
        //获取方法名
        String methodName = joinpoint.getSignature().getName();
        //获取相关参数
        Object[] arguments = joinPoint.getArgs();
        //生成类对象
        Class targetClass = Class.forName(targetName);
        //获取该类中的方法
        Method[] methods = targetClass.getMethods();

        String description = "";
        for(Method method : methods){
            if(!method.getName().equals(methodName)){
                continue;
            }
            Class[] clazz = method.getParameterTypes();
            if(clazz.length != arguments.length){
                continue;
            }
            description = method.getAnnotation(SystemServiceLog.class).description();
        }
        return description;
    }
 }

托管切点

因为这边是用的第二个方法，所以需要在applicationContext.xml内托管注解

<!-- 使用JtaTransactionManager 作为全局事物管理器 -->
<bean id="transactionManager" class="org.springframework.jdbc.datasource.DataSourceTransactionManager">
    <property name="dataSource" ref="dataSource"/>
</bean>
<!-- 启用声明式事物管理 -->
<tx:advice id="txAdvice" transaction-manager="transactionManager">
    <tx:attributes>
        <tx:method name="save*" propagation="REQUIRED"/><!--回滚方法名-->
        <tx:method name="*" read-only="true"/>
    </tx:attributes>
</tx:advice>
<aop:config>
    <aop:pointcut id="transactionPointcut"
        expression="execution(* com.base.service.impl1.*.*(..)||execution(* com.base.service.impl2.*.*(..))"/>
      <!-- order定义优先级 2 -->
    <aop:advisor pointcut-ref="transactionPointcut" advice-ref="txAdvice" order="2"/>
</aop:config>

<!-- 日志托管 -->
<bean id="SystemLogAspect" class="com.base.annotation.SystemLogAspect"></bean>
<!-- 申明切面类 -->
<!-- control层操作日志 -->
<aop:config>
    <!-- order定义优先级 3 -->
    <aop:aspect ref="SystemLogAspect" order="3">
        <aop:pointcut id="beforeMethods" expression="@annotation(com.base.annotation.SystemControllerLog)"/>
        <aop:before method="beforeMet" pointcut-ref="beforeMethods"/>
    </aop:aspect>
</aop:config>
<!-- service层异常日志 -->
<aop:config>
    <!-- order定义优先级 1 -->
    <aop:aspect ref="SystemLogAspect" order="1">
        <aop:pointcut id="afterThrowingMethods" expression="@annotation(com.base.annotation.SystemServiceLog)"/>
        <aop:after-throwing method="afterThrowingMet" pointcut-ref="afterThrowingMethods" throwing="e"/>
    </aop:aspect>
</aop:config>

总结

在完成的过程中出现了两个问题：
第一个是操作异常日志的时候，数据会回滚。找到原因是因为事物管理也扫描了全局impl包，导致异常日志在报错的时候将正常数据回滚的同时，也将异常日志插入的数据回滚了。这边的解决办法是定义优先级order，将异常日志定义为第一优先级。这样就避免了事物管理将异常日志数据回滚。
第二个问题是linux下，cglib代理无法插入异常日志，这里切换成jdk代理解决问题。
