后台方法

package com.base.util;
import java.io.BufferedInputStream;
import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.InputStream;

import com.artofsolving.jodconverter.DocumentConverter;
import com.artofsolving.jodconverter.openoffice.connection.OpenOfficeConnection;
import com.artofsolving.jodconverter.openoffice.connection.SocketOpenOfficeConnection;
import com.artofsolving.jodconverter.openoffice.converter.OpenOfficeDocumentConverter;
import com.artofsolving.jodconverter.openoffice.converter.
StreamOpenOfficeDocumentConverter;

public class DocConverter{
    /**
     * doc docx转换pdf
     */
    private static final int environment = 1;
    private String fileString;
    private String outputPath="";//输出路径
    private String fileName;
    private File pdfFile;
    private File swfFile;
    private File docFile;
    private boolean ispdf = false;

    public DocConverter(String fileString){
        ini(fileString);
    }

    /**
     * 重新设置file
     */
    public void setFile(String fileString){
        ini(fileString)
    }

    /**
     * 初始化
     */
    private void ini(String fileString){
        this.fileString = fileString;
        if(!"pdf".equals(fileString.substring(fileString.lastIndexOf(".")+1))){
            ispdf = true;
        }
        fileName = fileString.substring(0,fileString.lastIndexOf("."));
        docFile = new File(fileString);
        pdfFile = new File(fileName+".pdf");
        swfFile = new File(fileName+".swf");
    }

    /**
     * 转换pdf
     */
    publid void doc2pdf throws Exception{
        if(docFile.exists()){
            if(!pdfFile.exists()){
                OpenOfficeConnection connection = new SocketOpenOfficeConnection(8100);
                try{
                    connection.connect();
                    DocumentConverter converter = new StreamOpenOfficeDocimentConverter(connection);
                    converter.convert(docFile,pdfFile);
                    connection.disconnect();
                }catch(ConnectException e){
                    e.printStackTrace();
                    throw e;
                }catch(OpenOfficeException e){
                    e.printStackTrace();
                    throw e;
                }catch(Exception e){
                    e.printStackTrace();
                    throw e;
                }
            }else{

            }
        }else{

        }
    }

    /**
     * 转换swf
     */
    @SuppressWarnings("unused")
    private void pdf2swf() throws Exception{
        Runtime r = Runtime.getRuntime();
        if(!swfFile.exists()){
            if(pdfFile.exists()){
                if(environment == 1){
                    try{
                        Process p = r.exec("D:\pdf2swf.exe")+" "+
                        pdfFile.getPath()+" -o"+swfFile.getPath()+" -T 9";
                        loadStream(p.getInputStream());
                        loadStream(p.getErrorStream());
                        if(ispdf){
                            if(pdfFile.exists()){
                                pdfFile.delete();
                            }
                        }
                    }catch(IOException e){
                        e.printStackTrace();
                        throw e;
                    }
                }
            }else{
                System.out.println("pdf文件不存在");
            }
        }else{
            System.out.println("swf已存在");
        }
    }
    static String loadStream(InputStream in) throws IOException{
        int ptr = 0;
        in = new BufferedInputStream(in);
        StringBuffer buffer = new StringBuffer();
        while((ptr = in.read()) != -1){
            buffer.append((char)ptr);
        }
        return buffer.toString();
    }
    /**
     * 转换调用方法
     */
    @SuppressWarning("unused")
    public boolean conver(){
        if(swfFile.exists()){
            return true;
        }
        try{
            doc2pdf();
            pdf2swf();
        }catch(Exception e){
            e.printStackTrace();
            return false;
        }
        if(swfFile.exists()){
            return true;
        }else{
            return false;
        }
    }

    /**
     *返回文件路径 
     */
    public String getswfPath(){
        if(this.swfFile.exists()){
            String tempString = swfFile.getPath();
            tempString = tempString.replaceAll("\\\\","/");
            return tempString;
        }else{
            return "文件不存在";
        }
    }

    /**
     * 设置输出路径
     */
    public void setOutputPath(String outputPath){
        this.outputPath = outputPath;
        if(!"".equals(outputPath)){
            String realName = fileName.substring(fileName.lastIndexOf("/"),fileName.lastIndexOf("."));
            if(outputPath.charAt(outputPath.length()) == '/'){
                swfFile = new File(outputPath + realName + ".swf");
            }else{
                swfFile = new File(outputPath + realName + ".swf");
            }    
        }
    }
}
